/// Трейт для мод-специфичных знаний
pub trait ModOverride: Send + Sync {
    /// ID мода, для которого работает этот override
    fn mod_id(&self) -> &str;

    /// Какие конфиг-файлы создаёт мод и где
    fn config_locations(&self) -> Vec<ModConfigLocation>;

    /// Как найти настройку генерации руды в конфигах этого мода
    fn ore_gen_config_keys(&self) -> Vec<OreConfigMapping>;

    /// Дополнительные предметы, которые не видны в JAR
    /// (создаются программно)
    fn programmatic_items(&self) -> Vec<String>;

    /// Рецепты, которые мод создаёт программно (не через JSON)
    fn programmatic_recipes(&self) -> Vec<UnifiedRecipe>;
}

#[derive(Debug, Clone)]
pub struct ModConfigLocation {
    pub path: String,
    pub format: ConfigFileFormat,
    pub description: String,
}

#[derive(Debug, Clone)]
pub struct OreConfigMapping {
    pub resource_name: String,          // "copper"
    pub config_file: String,            // "config/mekanism/world.toml"
    pub enabled_key: String,            // "world.copper.shouldGenerate"
    pub vein_size_key: Option<String>,
    pub min_height_key: Option<String>,
    pub max_height_key: Option<String>,
}
